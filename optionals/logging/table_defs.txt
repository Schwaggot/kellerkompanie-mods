CREATE TABLE IF NOT EXISTS session (
	EventID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
	EventType ENUM('CONNECTED','DISCONNECTED'),
	PlayerUID CHAR(100) NOT NULL, 
	PlayerName CHAR(255) NOT NULL,
	JIP BOOLEAN NOT NULL,	 
	Time DATETIME NOT NULL) 
	ENGINE = MYISAM 
	CHARACTER SET 'utf8'
	COLLATE 'utf8_german2_ci';

CREATE TABLE IF NOT EXISTS mission (
	EventID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
	EventType ENUM('LOAD','START','END'),
	Time DATETIME NOT NULL,
	MissionName CHAR(255) NOT NULL,
	MissionVersion INT NOT NULL,
	WorldName CHAR(255) NOT NULL) 
	ENGINE = MYISAM 
	CHARACTER SET 'utf8'
	COLLATE 'utf8_german2_ci';
